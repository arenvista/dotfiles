% 6. CUSTOM BOXES (tcolorbox)
% ==========================================
\usepackage{tcolorbox}
\tcbuselibrary{breakable, skins}

% Base Custom Box Environment
\newenvironment{cbox}[2]{
    \begin{tcolorbox}[
        colback=white,
        colframe=gray,
        boxrule=0.5pt,
        arc=5pt,
        top=10pt,
        bottom=10pt,
        breakable, % Allows the box to break smoothly across pages
    ]
    \textcolor{#1}{\textbf{#2}} % Places the colored text at the start of the box
}{
    \end{tcolorbox}
}
